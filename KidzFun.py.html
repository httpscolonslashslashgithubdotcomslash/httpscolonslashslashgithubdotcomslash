def main():
    num_attending_fs = 0
    num_attending_ak = 0
    total_age_fs = 0
    total_age_ak = 0
    choice = 0

    while choice != 3:
        print("What do you want to do?")
        print()
        print("1 Enter data")
        print("2 Display current statistics")
        print("3 Exit the system")
        print()
        print("=" * 80)
        try:
            choice = check_number("Enter your choice "
                                  "(number between 1 and 3): ")
            if choice == 1:
                data = enter_data()

                if data[0] == "F":
                    total_age_fs += data[1]
                    num_attending_fs += 1

                elif data[0] == "A":
                    total_age_ak += data[1]
                    num_attending_ak += 1

            elif choice == 2:
                statistics(num_attending_fs, num_attending_ak, total_age_fs, total_age_ak)

            elif choice == 3:
                confirm_exit = input("Are you sure you want to exit? All data "
                                     "will be lost!!\nEnter 'Y' to confirm or "
                                     "any other key to go back: ").upper()
                if confirm_exit == "Y":
                    statistics(num_attending_fs, num_attending_ak, total_age_fs, total_age_ak)
                    end_program()
                    break
                else:
                    choice = 0
            else:
                print("Must be an integer between 1 and 3")
        except ValueError:
            print("Not a valid choice!")


def check_number(text):
    valid = False
    while not valid:
        try:
            number = int(input(text))
            if isinstance(number, int):
                valid = True
                return number
        except ValueError:
            print("Woops...")


def enter_data():
    program = input("\nWhich programme did the child attend?\n"
                    "'F' for Fun in the Sun; or\n"
                    "'A' for Active Kids; or \n"
                    "any other key to go back: ").upper()
    if program == "A" or program == "F":
        question = "\nInput the child's age (in years)\n"\
                   "NOTE: age must be between 5 and 15: "
        age = check_number(question)
        while not 5 <= age <= 15:
            age = check_number(question)
        return program, age
    else:
        print()
        return "", 0


def statistics(num_attending_fs, num_attending_ak, total_age_fs, total_age_ak):
    if num_attending_fs == 0:
        print("\nThe average for the 'Fun in the Sun' programme could not be "
              "calculated because no data has been entered for this programme")

    else:
        average_age_fs = total_age_fs / num_attending_fs
        print(f"\n The average age of children in the 'Active Kids programme "
              f"is {round(average_age_fs)}")
        print(f"The total number registered for 'Active Kids is "
              f"{num_attending_fs}")

    if num_attending_ak == 0:
        print("\nThe average for the 'Active Kids' programme could not be "
              "calculated because no data has been entered for this programme")
    else:
        average_age_ak = total_age_ak / num_attending_ak
        print(f"\n The average age of children in the 'Active Kids programme "
              f"is {round(average_age_ak)}")
        print(f"The total number registered for 'Active Kids is "
              f"{num_attending_ak}")


def end_program():
    print("\nGoodbye")


main()
